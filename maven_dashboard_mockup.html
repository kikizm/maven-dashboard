BigQuery SQL Queries
Q1 — Revenue by Year 
SELECT
  EXTRACT(YEAR FROM o.created_at) AS year,
  COUNT(DISTINCT o.order_id)     AS total_orders,
  ROUND(SUM(o.price_usd), 2)        AS revenue,
  ROUND(AVG(o.price_usd), 2)        AS avg_order_value,
  ROUND((SUM(o.price_usd)
    - SUM(o.cogs_usd))
    / SUM(o.price_usd) * 100, 2)   AS gross_margin_pct
FROM orders o
GROUP BY year
ORDER BY year;
Q2 — Monthly MoM Growth 
SELECT
  FORMAT_DATE('%Y-%m', created_at) AS month,
  COUNT(DISTINCT order_id)         AS orders,
  ROUND(SUM(price_usd), 2)           AS revenue,
  ROUND(
    (SUM(price_usd)
     - LAG(SUM(price_usd)) OVER (
         ORDER BY FORMAT_DATE('%Y-%m', created_at)))
    / NULLIF(LAG(SUM(price_usd)) OVER (
         ORDER BY FORMAT_DATE('%Y-%m', created_at)),0)*100
  , 2) AS mom_growth_pct
FROM orders
GROUP BY month  ORDER BY month;
Q3 — Product Refund Analysis 
SELECT
  p.product_name,
  COUNT(DISTINCT oi.order_item_id)    AS units_sold,
  ROUND(SUM(oi.price_usd), 2)          AS gross_revenue,
  COUNT(DISTINCT r.order_item_refund_id)
                                         AS refunds,
  ROUND(
    COUNT(DISTINCT r.order_item_refund_id)
    * 100.0 / COUNT(DISTINCT oi.order_item_id)
  , 2)                                  AS refund_rate_pct
FROM order_items oi
JOIN products p ON oi.product_id = p.product_id
LEFT JOIN order_item_refunds r
  ON oi.order_item_id = r.order_item_id
GROUP BY p.product_name
ORDER BY gross_revenue DESC;
Q4 — Net Revenue After Refunds 
SELECT
  FORMAT_DATE('%Y-%m', o.created_at)  AS month,
  ROUND(SUM(o.price_usd), 2)
                  AS gross_revenue,
  ROUND(SUM(COALESCE(r.refund_amount_usd,0)),2)
                  AS refunds,
  ROUND(
    SUM(o.price_usd)
    - SUM(COALESCE(r.refund_amount_usd,0))
  , 2)            AS net_revenue
FROM orders o
LEFT JOIN order_item_refunds r
  ON o.order_id = r.order_id
GROUP BY month
ORDER BY month;
